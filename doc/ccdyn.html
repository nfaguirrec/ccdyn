 &data
        nAtoms = 301
        initialDensity = "density.N300.dat"
        actions = "STSEPFS"
/

&save
        geometry = "geom.xyz0", "geom.xyz1", "geom.xyz2"
        adf = "geom.adf0", "geom.adf1", "geom.adf2"
/

&thermalize
        nSteps = 1000
        timeStep = 0.05
        temperature = 0.1
        tau = 0.4
        showFrequency = 100
        geometryHistoryFile = "hist.xyz"
        velocityHistoryFile = "hist.vxyz"
        integrator = "VelocityVerlet"
/

&freeEvolution
        nSteps = 12000
        timeStep = 0.005
        showFrequency = 400
        geometryHistoryFile = "hist.xyz"
        velocityHistoryFile = "hist.vxyz"
        integrator = "VelocityVerlet"
/

&pushMolecule
        velocity = 0.0, 0.0, -1.0
/

&externalField
        targetAtom = "He", "Au"
        origin = 0.0,0.0,-30.0,  0.0,0.0,-30.0
        name = "HeTiO2_M3DP", "AuTiO2_SMSP"
/

Las líneas que comienzan con ! o espacio en blanco son ignoradas. El fichero de entrada es sensible a las mayusculas.
